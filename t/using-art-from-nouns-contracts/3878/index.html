import { promises as fs } from 'fs';
import path from 'path';

const encode = async () => {
  const encoder = new PNGCollectionEncoder();

  const partfolders = ['1-bodies', '2-accessories', '3-heads', '4-glasses'];

    const folderpath = path.join(__dirname, '../images/v0', folder);
